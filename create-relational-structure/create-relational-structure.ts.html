export function createRelationalStructure(

  parentTable?: RelationalTable,
): RelationalTable[] {

  const nestedData: Map<string, NormalizedData> = new Map();

  });

        }

        value.forEach((item) =>
          nestedDataByKey.push({ ...item, normalize_parent_id }),
        );
